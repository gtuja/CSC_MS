63HAL_StatusTypeDef HAL_InitTick(uint32_t TickPriority)
64{
65 RCC_ClkInitTypeDef clkconfig;
66 uint32_t uwTimclock, uwAPB1Prescaler;
67
68 uint32_t uwPrescalerValue;
69 uint32_t pFLatency;
70 HAL_StatusTypeDef status = HAL_OK;
71
72 /* Enable TIM6 clock */
73 __HAL_RCC_TIM6_CLK_ENABLE();
74
75 /* Get clock configuration */
76 HAL_RCC_GetClockConfig(&clkconfig, &pFLatency);
77
78 /* Get APB1 prescaler */
79 uwAPB1Prescaler = clkconfig.APB1CLKDivider;
80 /* Compute TIM6 clock */
81 if (uwAPB1Prescaler == RCC_HCLK_DIV1)
82 {
83 uwTimclock = HAL_RCC_GetPCLK1Freq();
84 }
85 else
86 {
87 uwTimclock = 2UL * HAL_RCC_GetPCLK1Freq();
88 }
89
90 /* Compute the prescaler value to have TIM6 counter clock equal to 1MHz */
91 uwPrescalerValue = (uint32_t) ((uwTimclock / 1000000U) - 1U);
92
93 /* Initialize TIM6 */
94 htim6.Instance = TIM6;
95
96 /* Initialize TIMx peripheral as follow:
97
98 + Period = [(TIM6CLK/1000) - 1]. to have a (1/1000) s time base.
99 + Prescaler = (uwTimclock/1000000 - 1) to have a 1MHz counter clock.
100 + ClockDivision = 0
101 + Counter direction = Up
102 */
103 htim6.Init.Period = (1000000U / 1000U) - 1U;
104 htim6.Init.Prescaler = uwPrescalerValue;
105 htim6.Init.ClockDivision = 0;
106 htim6.Init.CounterMode = TIM_COUNTERMODE_UP;
107 htim6.Init.AutoReloadPreload = TIM_AUTORELOAD_PRELOAD_DISABLE;
108
109 status = HAL_TIM_Base_Init(&htim6);
110 if (status == HAL_OK)
111 {
112 /* Start the TIM time Base generation in interrupt mode */
113 status = HAL_TIM_Base_Start_IT(&htim6);
114 if (status == HAL_OK)
115 {
116 /* Enable the TIM6 global Interrupt */
117 HAL_NVIC_EnableIRQ(TIM6_DAC_LPTIM1_IRQn);
118 /* Configure the SysTick IRQ priority */
119 if (TickPriority < (1UL << __NVIC_PRIO_BITS))
120 {
121 /* Configure the TIM IRQ priority */
122 HAL_NVIC_SetPriority(TIM6_DAC_LPTIM1_IRQn, TickPriority, 0U);
123 uwTickPrio = TickPriority;
124 }
125 else
126 {
127 status = HAL_ERROR;
128 }
129 }
130 }
131
132 /* Return function status */
133 return status;
134}
