218void SystemCoreClockUpdate(void)
219{
220 uint32_t tmp;
221 uint32_t pllvco;
222 uint32_t pllr;
223 uint32_t pllsource;
224 uint32_t pllm;
225 uint32_t hsidiv;
226
227 /* Get SYSCLK source -------------------------------------------------------*/
228 switch (RCC->CFGR & RCC_CFGR_SWS)
229 {
230 case RCC_CFGR_SWS_0: /* HSE used as system clock */
231 SystemCoreClock = HSE_VALUE;
232 break;
233
234 case (RCC_CFGR_SWS_1 | RCC_CFGR_SWS_0): /* LSI used as system clock */
235 SystemCoreClock = LSI_VALUE;
236 break;
237
238 case RCC_CFGR_SWS_2: /* LSE used as system clock */
239 SystemCoreClock = LSE_VALUE;
240 break;
241
242 case RCC_CFGR_SWS_1: /* PLL used as system clock */
243 /* PLL_VCO = (HSE_VALUE or HSI_VALUE / PLLM) * PLLN
244 SYSCLK = PLL_VCO / PLLR
245 */
246 pllsource = (RCC->PLLCFGR & RCC_PLLCFGR_PLLSRC);
247 pllm = ((RCC->PLLCFGR & RCC_PLLCFGR_PLLM) >> RCC_PLLCFGR_PLLM_Pos) + 1UL;
248
249 if(pllsource == 0x03UL) /* HSE used as PLL clock source */
250 {
251 pllvco = (HSE_VALUE / pllm);
252 }
253 else /* HSI used as PLL clock source */
254 {
255 pllvco = (HSI_VALUE / pllm);
256 }
257 pllvco = pllvco * ((RCC->PLLCFGR & RCC_PLLCFGR_PLLN) >> RCC_PLLCFGR_PLLN_Pos);
258 pllr = (((RCC->PLLCFGR & RCC_PLLCFGR_PLLR) >> RCC_PLLCFGR_PLLR_Pos) + 1UL);
259
260 SystemCoreClock = pllvco/pllr;
261 break;
262
263 case 0x00000000U: /* HSI used as system clock */
264 default: /* HSI used as system clock */
265 hsidiv = (1UL << ((READ_BIT(RCC->CR, RCC_CR_HSIDIV))>> RCC_CR_HSIDIV_Pos));
266 SystemCoreClock = (HSI_VALUE/hsidiv);
267 break;
268 }
269 /* Compute HCLK clock frequency --------------------------------------------*/
270 /* Get HCLK prescaler */
271 tmp = AHBPrescTable[((RCC->CFGR & RCC_CFGR_HPRE) >> RCC_CFGR_HPRE_Pos)];
272 /* HCLK clock frequency */
273 SystemCoreClock >>= tmp;
274}
