227void MessageQueueProducer(void *argument)
228{
229 /* USER CODE BEGIN 5 */
230 /* Infinite loop */
231 for (;;)
232 {
233 if (osMessageQueuePut(osQueueHandle, &ProducerValue, 100, 0) != osOK)
234 {
235 ++ProducerErrors;
236
237 /* Switch On continuously LED4 to indicate error */
238 BSP_LED_On(LED4);
239 }
240 else
241 {
242 /* Increment the variable we are going to post next time round. The
243 consumer will expect the numbers to follow in numerical order */
244 ++ProducerValue;
245
246 if( (ProducerErrors == 0) && (ConsumerErrors == 0) )
247 BSP_LED_Toggle(LED4);
248 osDelay(1000);
249 }
250 }
251 /* USER CODE END 5 */
252}
